# 题目

给定一个只包含数字的字符串，复原它并返回所有可能的 IP 地址格式。

示例: 输入: "25525511135" 输出: ["255.255.11.135", "255.255.111.35"]

来源：力扣（LeetCode）

# 思路

采用递归思想本质

  • 每一个处理过程是相同的
  • 输入输出是相同的
  • 处理次数未知

# 代码

/**
 * @param {string} s
 * @return {string[]}
 */

var restoreIpAddresses = function(s) {
    let arr=[];//保存有效的ip
    //递归思想，将字符串切割，并验证这种切割是否合法
    //current存储分割的每一部分 old剩余部分字符串的长度
    let findIp=(current,old)=>{
        //长度是4并且还原之后等于原字符串，递归结束
        if(current.length===4&&current.join('')===s){
            arr.push(current.join('.'));
        }else{
            for(let i=0,len=Math.min(3,old.length);i<len;i++){
                let cur_slice=old.substr(0,i+1);
                //切割一次，剩下几次切割，递归完成 如果发现切下来的值比256大，那这个就不是ip，废了他
                const exp=/^0\d+/g;
                if(cur_slice<256&&!exp.test(cur_slice)){
                    findIp(current.concat([cur_slice]),old.substr(i+1));
                }
            }
        }
    }
    findIp([],s);
    return arr;
};
