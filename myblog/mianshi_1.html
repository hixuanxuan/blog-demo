//class写法
class Animal{
    constructor(name){
        this.name=name;
    }
    eat(){
        console.log(this.name+'animal eat')
    }
}
//有继承必须写super
//否则不能继承父类的属性
class Dog extends Animal{
    constructor(name) {
        super(name);
        this.name = name;
    }
    bark (){
        console.log(this.name+'dog bark');
    }
}
let labuladuo = new Dog('xiaoming');
labuladuo.eat();//xiaominganimal eat
labuladuo.bark();//xiaomingdog bark

  • Promise的基本使用和原理 callback hell Promise语法 下面是一个简单的，图片加载返回图片dom的例子，异步读取图片，返回dom，获取dom宽度，如果采用普通的函数处理代码会很难读。
function loadImg(src) {
    const promise = new Promise((resolve, reject) => {
        let img = new Image();
        img.onload = () => {
            // console.log('onload')
            resolve(img);
        }
        img.onerror = () => {
            reject();
        }
        img.src = src;
    });
    return promise;
}
let src = 'http://a0.att.hudong.com/16/12/01300535031999137270128786964.jpg';
let result = loadImg(src);
result.then((img) => {
    console.log(img.width,img.height);
}, (err) => {
    console.log('failed');
})
