# 观察者模式

# 1.1 示例代码

// 主题 保存状态，状态变化之后触发所有观察者对象
class Subject{
    constructor() {
        this.state = 0;
        this.observers = [];
    }
    getState() {
        return this.state;
    }
    setState(newState) {
        this.state = newState;
        this.notifyAllObservers();
    }
    notifyAllObservers() {
        this.observers.forEach((observer) => {
            observer.update();
        })
    }
    //添加观察者
    attach(observer) {
        this.observers.push(observer);
    }
}
//观察者
class Observer{
    constructor(name, subject) {
        this.name = name;
        this.subject = subject;
        this.subject.attach(this);
    }
    //等待被触发，等待被调用
    update() {
        console.log(`${this.name}update,state:${this.subject.getState()}`)
    }
}

let s = new Subject();
let o1 = new Observer('o1', s);
let o2 = new Observer('o2', s);

function loadImg(src) {
    return new Promise((resolve, reject) => {
        let img = new Image();
        img.onload = () => {
            // console.log('onload')
            resolve(img);
        }
        img.onerror = () => {
            reject();
        }
        img.src = src;
    });
}

# 1.2.3 nodejs自定义事件

EventEmitter，自定义事件，on，emit

const EventEmitter = require('events').EventEmitter;
// 任何构造函数都可以继承EventEmitter的构造方法，on emit
class Dog extends EventEmitter{
   constructor(name) {
       super();
       this.name = name;
   }
}
var simon = new Dog('simon');
// 注意这个cbs使用了箭头函数this的指向就不对了
simon.on('bark', function() {
   console.log(this);
   console.log(this.name + 'barked');
})
setInterval(() => {
   simon.emit('bark');
}, 1000);

# 1.4 vue中的设计模式

Vue中用到了观察者模式和发布订阅模式

Observer（监听者）监听数据变化，通知观察者列表，Dep对象观察者列表，Watcher是观察者实例，watcher用于订阅update以及添加观察者。源是data，结果是view，data和view组成vm。

Observer在Vue实例化调用的，data遍历一遍，然后调用defineReactive 数据读取：添加观察者（依赖收集），数据变化：通知观察者。 实例的update函数，在Watcher里面建立的， Dep.target观察者实例，就是watcher 监听对象的时候，判断Dep.target是不是有值，他就是观察者实例 watcher的get里面会现缓存自己，然后判断观察者列表中有没有自己（利用data[shuxing]这样会调用对象的getter，强制执行监听器里面的get函数，get里面会把当前对象和Observer列表发生关联），如果没有那就添加，然后释放自己。

修改数据的时候，触发Observer的set，set会通知观察者列表，执行update，update也就是Watcher的run，也就是回调，回调执行view，让视图更新。

Complile 根据模版，创建dom，绑定数据，双向数据流和单向的分开。编译模版：CompileElement，判断结点的类型，普通结点，单向绑定，遇到v-model双向绑定的特殊处理。
