串联所有子串

/**
 * @param {string} s
 * @param {string[]} words
 * @return {number[]}
 */
var findSubstring = function(s, words) {
    let arr=[];//存储需要的子串
    //last 上一次的处理结果 remain 剩下的数组
    let range=(last,remain)=>{
        // console.log(last)
        //边界条件，数组里面的每一项都遍历完了，都踢出去了
        if(last.length===words.length){
            arr.push(last);
        }else{
            //如果不是，把没处理的数组遍历一遍
            remain.forEach((item,index)=>{
                 // console.log(last)
                //复制数组remain
                let tmp=[].concat(remain);
                tmp.splice(index,1);
                //把删除的值处理成一个数组
                range(last.concat(item),tmp);
            })
        }
    }
    range([],words);
   // console.log(arr);
    //注意匹配不到会返回-1，要去除-1
    // return Array.from(new Set(arr.map(item=>s.indexOf(item.join(''))).filter((item)=>item!=-1).sort()));
    let num=[];
    for(let i=0,len=s.length;i<len;i++){
        arr.forEach((item)=>{
            if(s.substr(i).indexOf(item.join(''))!=-1){
                console.log(item.join(''))
                if(item.length!==0){
                   num.push(i+s.substr(i).indexOf(item.join('')));;
                }
            }
        })
    }
    return Array.from(new Set(num)).sort();
};
