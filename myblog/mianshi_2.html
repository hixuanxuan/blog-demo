# 原型

谈论原型，用jquery库来学习再好不过了。 实例不同，构造函数是一个，构造函数原型上面的方法，都可以被继承。 当多个实例可以调用相同的方法，说明方法在构造函数上面。

简单的写一下zepto的构造函数,这只是一个例子，实际实现肯定要比这个复杂的多。

(function (window) {
    var zepto = {};
    zepto.init = function (selector) {
        var slice = Array.prototype.slice;
        var dom = slice.call(document.querySelectorAll(selector));//类数组变成数组
        return zepto.Z(dom, selector);
    }
    zepto.Z = function (dom, selector) {
        return new Z(dom,selector);
    }
    function Z(dom, selector) {
        var i ,len = dom ? dom.length : 0;
        for (i = 0; i < len; i++){
            //把dom里面每一个元素赋值到Z的属性中
            this[i] = dom[i];
        }
        this.length = len;
        this.selector = selector || '';
    }
    //首先，挂载到全局也就是window上的$，如何实现无new操作
    //$的执行，会返回zepto的init
    //zepto对象的init，会把select到的dom保存，并返回zepto的Z方法，
    //zepto的Z方法返回new Z，也就是会执行Z的构造函数，
    //在Z的构造函数里面，将dom挂在到当前new的Z
    var $ = function (selector) {
        return zepto.init(selector);
    };
    window.$ = $;
    //构造函数的原型
    zepto.Z.protoType = $.fn = Z.prototype = {
        constructor:zepto.Z,
        html: function () {
            console.log(this[0])//这也就是为什么jquery[0]转化dom的原因
        }
    }

})(window)

jquery如何使用原型

/**
 * 对jq的原型方法简写
 */
var jQuery = function (selector) {
    return new jQuery.fn.init(selector);
}


//定义JQ的构造函数
var init = jQuery.fn.init = function (selector) {
    var dom = Array.prototype.slice.call(document.querySelectorAll(selector));
    let i, len = dom ? dom.length : 0;
    for (i = 0; i < len; i++){
        this[i] = dom[i];
    }
    this.length = len;
    this.selector = selector || '';
}
jQuery.fn= jQuery.prototype = {
    //为什么将原型同时赋值给jQuery.fn和构造函数的原型
    constructor: jQuery,
    //其他函数...
    css: function (key,value) {
        
    },
    html: function(value){
        
    }
};
//定义构造函数的原型
init.prototype = jQuery.fn;
window.$ = jQuery;

扩展性

为什么要把原型方法放在$.fn上面

为了扩展 $.fn.getNodeName=function(){return this[0].nodeName}

好处：只有$会暴露给window全局变量 不能通过构造函数扩展 将插件统一扩展到$.fn.xxx这个接口，方便使用。
